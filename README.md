SERVER:
-OPEN CMD
    -GO TO PARTFINDER FOLDER
    -WRITE "npm install dotenv"
    -ADD THE ".env" file into a folder named "misc"
    -WRITE "node index.js"

CLIENT:
-OPEN CMD
    -GO TO PARTFINDER/STATIC FOLDER
    -WRITE "npm install"
    -WRITE "npm run dev"
    -OPEN THE BROWSER AND SEARCH FOR THE SITE WRITTEN IN THE CMD
